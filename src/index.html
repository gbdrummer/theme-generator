import String from ‘./String.js’

      /*
       * class CustomString
       * Test class for theme demonstration.
       */
      class CustomString extends String {
        #contents

        constructor (string) {
          this.#contents = string  
        }

        get contents () {
          return this.#contents
        }

        validate (string) {
          let valid = true

          if (string === undefined || string === null || string !== this.#contents) {
            valid = false
            console.log(`string “${string}” does not match “${this.#contents}”`)
          }

          return valid
        }
      }
